## Install

```bash
npm install --save react-logic-stack
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-logic-stack'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
